    // Fill Theory Quantum Decimal System
    class QuantumDecimal {
        constructor(value, precision = 12) {
            this.value = parseFloat(value.toFixed(precision));
            this.precision = precision;
        }
        
        shiftDecimal(n) {
            let strVal = this.value.toString().replace('.', '');
            if (n > 0) {
                strVal = strVal.padEnd(strVal.length + n, '0');
            } else if (n < 0) {
                strVal = '0'.repeat(-n) + strVal;
            }
            let newVal = parseFloat(strVal) / Math.pow(10, strVal.length - n);
            this.value = parseFloat(newVal.toFixed(this.precision));
            return this.value;
        }
        
        toAngle() {
            return this.value % 360;
        }
    }

    function generatePiPatch(iterations = 50, microShift = 0.001, precision = 12) {
        const shapes = ["triangle", "square", "pentagon", "circle", "spiral"];
        const constants = {
            pi: 3.141592653589793,
            phi: 1.618033988749895,
            sqrt2: 1.4142135623730951,
            sqrt3: 1.7320508075688772,
            e: 2.718281828459045
        };
        
        let patch = [];
        
        for (let i = 0; i < iterations; i++) {
            shapes.forEach((shape, shapeIndex) => {
                // Apply quantum decimal precision to angle calculation
                let baseTheta = (i * constants.pi * (shapeIndex + 1)) % 360;
                let qd = new QuantumDecimal(baseTheta, precision);
                let theta = (qd.toAngle() + microShift * 360) % 360;
                
                patch.push({
                    shape: shape,
                    theta: theta,
                    value: theta / 360,
                    quantumDecimal: qd,
                    iteration: i,
                    shapeIndex: shapeIndex
                });
            });
        }
        
        return patch;
    }

    // Helix Web Generator
    function createHelix(node, turns = 5, microSpin = 0.001, ultraSpin = 0.00000000001) {
        const curve = new THREE.Curve();
        curve.getPoint = function(t) {
            const angle = t * Math.PI * 2 * turns + node.userData.value * microSpin;
            const radius = 2 + node.userData.value * 0.5;
            const y = t * 5 + node.userData.value * ultraSpin;
            return new THREE.Vector3(
                Math.cos(angle) * radius,
                y,
                Math.sin(angle) * radius
            );
        };
        
        const geometry = new THREE.TubeGeometry(curve, 64, 0.02, 8, false);

            color: 0x000000,
            wireframe: true,
            transparent: true,
            opacity: 0.7
        });
        
        const mesh = new THREE.Mesh(geometry, material);
        mesh.position.copy(node.position);
        return mesh;
    }
    
    // Parallel Fill Grid Function
    function fillGrid(nodes, maxSteps = 50) {
        const grid = new Map();
        const fillGeometry = new THREE.BufferGeometry();
        const positions = [];
        
        nodes.forEach(node => {
            let steps = 0;
            let position = node.position.clone();
            const delta = node.userData.value * 0.2;
            
            while (steps < maxSteps) {
                const key = `${position.x.toFixed(3)},${position.y.toFixed(3)},${position.z.toFixed(3)}`;
                if (!grid.has(key)) {
                    grid.set(key, node.userData.value);
                    positions.push(position.x, position.y, position.z);
                    
                    position.x += delta;
                    position.y += delta * 0.5;
                    position.z += delta;
                } else {
                    break; // collision → sealed
                }
                steps++;
            }
        });
        
        if (positions.length > 0) {
            fillGeometry.setAttribute('position', new THREE.Float32BufferAttribute(positions, 3));
            const fillMaterial = new THREE.PointsMaterial({
                color: 0x000000,
                size: 0.05,
                transparent: true,
                opacity: 0.3
            });
            return new THREE.Points(fillGeometry, fillMaterial);
        }
        return null;
    }
